\underline{Objectif client :}
	\begin{itemize}
		\item Mise en valeur des fonctionnements des produits Intel
	\end{itemize}

\underline{Objectifs du projet :}
	\begin{itemize}
		\item Mettre en oeuvre les fonctionnalités d'un drone de vol paramoteur :
			\item[$\bullet$] Gestion d'autonomie du drone
			\item[$\bullet$] Maitrise des technologies appliquées au drone
			\item[$\bullet$] Fonctionnement du drone à établir : Detection d'une voiture
		\end{itemize}

\underline{Livrables principaux :}
	\begin{itemize}
		\item materiel :
			\item[$\bullet$] Drone de vol paramoteur
			\item[$\bullet$] Telephone avec camera, fonction GPS intégrée
			\item[$\bullet$] caméra GoPro
			\item[$\bullet$] détecteur de vent
		\item développement applicatif :
			\item[$\bullet$] plateform d'environnement de simulation
			\item[$\bullet$] gestion de connexion réseau 3G, wifi
			\item[$\bullet$] gestion de suivi de voiture (tracking) 
		\end{itemize}
	
	\underline{Livrables intermédiaires}
		\begin{itemize}
				\item[$\bullet$] blog contenant les sources et la documentation du projet
		\end{itemize}
		
	\underline{risques}
			\begin{itemize}
				\item[$\bullet$] pas d'utilisation de licenses appropriées au drone 
				\item[$\bullet$] légalisation du vol de drone sur certaines zones (pas de permis de vol)
				\item[$\bullet$] qualité non optimale de l'image lors de la capture d'image.
				\item[$\bullet$] problème de précision lors de la direction du drone
				\item[$\bullet$] possibilité de rencontre de perturbations et d'obstacles dans l'environnement durant le vol
		\end{itemize}
		
		\underline{solution}
			\begin{itemize}
				\item[$\bullet$] utilisation de license libre
				\item[$\bullet$] vol sur secteur restreint (champs)
				\item[$\bullet$] gestion du rendu visuel par intermédiaire du GoPro
				\item[$\bullet$] investissement principal sur l'environnement de simulation
			\end{itemize}
